Уменьшение связанности

  • Классы А и Б – связаны тогда, когда нельзя понять как А устроен, не зная устройства Б.

Связанности мало, значит меньше нужно помнить, значит легче понять код.
Но связанность уменьшается ‘переходниками’.
Больше ‘переходников’, значит больше кода, значит труднее понять код.

Жажда скорости

‘Переходники’ уменьшают производительность игры, но ускоряют прототипирование.
Легче сделать интересную игру быстрой (убрать ‘переходники’), чем быструю игру интересной.
Говнокод пишется ещё быстрее чем код с ‘переходниками’,
поэтому, его можно использовать для прототипирования.

  • Используем шаблоны, пока не получится что-то интересное, а затем оптимизируем.
  • Прототипный говнокод, после проверки идей, необходимо выбрасывать.


Хорошо оптимизированный код часто

  • прост для восприятия
  • покрывает малую область использования
  • долго придумывается (от структур, через алгоритм)
  • краток


Элегантный код – неплохо оптимизирован и устойчив (покрывает большую облать использования).

bool closed = false;
inline void close () {closed = true;}

struct Command {
    virtual void exec (Actor& a) = 0;
    virtual ~Command() {}
};

// Новые команды добавляются совсем просто
struct NullObjCommand: public Command {void exec (Actor& a) override {}};
struct JumpCommand:    public Command {void exec (Actor& a) override {a.jump ();}};
struct SitDownCommand: public Command {void exec (Actor& a) override {a.sit ();}};
struct CloseCommand:   public Command {void exec (Actor& a) override {close ();}};

struct InputHandler
{
    std::unique_ptr<Command> button_W, button_S, button_N, button_Q;
    Command& handleInput (Key pressed) {
        switch (pressed) {
            // Новые клавиши/события добавляются чисто механически

            case Key::S: return button_S.get ();
            case Key::N: return button_N.get ();
            case Key::Q: return button_Q.get ();
            default return pass.get ();
        }
    }
private:
    static std::unique_ptr<Command> pass;
};
std::unique_ptr<Command> InputHandler::pass {new NullObjCommand ()};

    Texture texture_;

struct World {

    World () {
        for (size_t x=0, x < width, ++x) {     // рандомно разбрасываем
            for (size_t y=0, y < width, ++y) { // горы и траву по карте
                if (random (10) == 0) {tiles_[x][y] = &hillTerrain_;}
                else {tiles[x][y] = &grassTerrain_;}
            }
        }
        size_t x = random (width);
        for (size_t y=0, y < width, ++y)  // где-то в центре,
            tiles[x][y] = &riverTerrain_; // добавляем реку

    Terrain

    Terrain* tiles_[width][height];

- Observer

enum Event{
    THE_ROOF_IS_IN_FIRE,
    SOME_ANOTHER_EVENT
};

class Observer {
public:
    virtual ~Observer(){}
    virtual void OnNotify(Event event) = 0;
};

class Subject {
protected:
    void notify(Event event) {
        for (auto& it : observers) it->OnNotify (event);
    }
    std::vector<Observer*> observers;
};

class Achievements : public Observer { // наблюдатель
public:
    virtual void OnNotify(Event event) {
        switch(event){ // реагируем на события
            case Event::THE_ROOF_IS_IN_FIRE:
                if (!fireAchivementUnlocked) {
                    puts("[We don't need no water, let the mfkr burn!] Achievement unlocked");
                    fireAchivementUnlocked = true;
                }
            break;
            default: break;
        }
    }
private:
    bool fireAchivementUnlocked = false;
};

class Physics : public Subject // наблюдаемый
{
public:
    void UpdateFire() {
        if (random(10) > 8) notify (Event::THE_ROOF_IS_IN_FIRE);
    }
    void Subscribe(Observer* o){observers.push_back(o);}
};

auto main() -> int {
    Achievement achivement;
    Physics roof;
    roof.Subscribe (&achivement);
    // 1. можно подписать ачивку на кучу других вещей
    // 2. можно подписать на горящую крышу, воспроизведение звука, анимации и т.д.

    for (size_t idx = 0; idx < 1000; ++idx) {
        roof.UpdateFire ();
    }
}
